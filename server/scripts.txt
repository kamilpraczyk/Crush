//Create table
CREATE TABLE Users (
  user_id INT NOT NULL AUTO_INCREMENT,
  email VARCHAR(80) NOT NULL,
  name VARCHAR(50) NOT NULL,
  password CHAR(41) NOT NULL,
  registered CHAR(20) NOT NULL,
  valid_to CHAR(20) NOT NULL,
  PRIMARY KEY (user_id),
  UNIQUE INDEX (email)
);
